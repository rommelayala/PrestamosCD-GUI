DROP TABLE `discos`;

CREATE TABLE `discos` (
  `id` mediumint(8) unsigned NOT NULL auto_increment,
  `titulo` varchar(255),
  `autor` varchar(255) default NULL,
  `genero` varchar(100) default NULL,
  `estado` varchar(10),
	`prestamo` varchar(250),
  PRIMARY KEY (`id`)
) AUTO_INCREMENT=1;
